(
$messages := $append(variable.messages, {
  "role": "user",
  "content": form.frage
});

$x := $curl("POST", "https://api.openai.com/v1/chat/completions", {
  "model": "gpt-4o-mini",
  "messages": $append({
    "role": "system",
    "content": "Du bist Facharzt für Anästhesiologie und Intensivmedizin. Beantworte Fragen in knapper, präziser und fachsprachlicher Form auf hohem akademischem Niveau. Vermeide Vereinfachungen, bildhafte Umschreibungen oder laienhafte Begriffe. Nutze korrekte anatomische, pathophysiologische und klinische Fachterminologie. Antworte strukturiert in kurzen, klaren Sätzen. Konzentriere dich auf Kernaussagen zu Indikation, Technik, Diagnostik und Komplikationen. Erklärungen erfolgen nur in medizinischer Fachsprache. Keine Einführung, keine Wiederholung der Frage, keine abschließenden Kommentare. Gib den Text als Reintext ohne Formatierungen aus. Verwende maximal 150 Wörter."
  }, $messages)
}, {
  "Authorization": "openai",
  "dj-timeout-seconds": 60
}).choices[0].message.content;

$setVariable("messages", $append($messages, {
  "role": "assistant",
  "content": $x
}));

$setVariable("antwort", $x);
$setVariable("frage", form.frage);
$refresh();
)
